Гра для 2 гравців

import pygame

import sys

import random

pygame.init()

pygame.mixer.init()

info = pygame.display.Info()

WIDTH, HEIGHT = info.current_w, info.current_h

screen = pygame.display.set_mode((WIDTH, HEIGHT), pygame.FULLSCREEN)

pygame.display.set_caption("Доганялки зі стрільбою")

WHITE = (255, 255, 255)

RED = (255, 0, 0)

BLUE = (0, 0, 255)

BLACK = (0, 0, 0)

GRAY = (200, 200, 200)

WALL_COLOR = (100, 100, 100)

PORTAL_COLOR = (0, 255, 255)

player_size = 50

speed = 5

bullet_speed = 10

portal_cooldown = 3000 # milliseconds

num_portal_pairs = 3

font = pygame.font.SysFont(None, 72)

small_font = pygame.font.SysFont(None, 36)

button_width, button_height = 300, 80

button_rect = pygame.Rect((WIDTH - button_width) // 2, (HEIGHT - button_height) // 2 + 100, button_width, button_height)

walls = [1] # Стін більше немає — замість лабіринту буде відкритий ландшафт

def get_random_spawn():

while True:

x = random.randint(0, WIDTH - player_size)

y = random.randint(0, HEIGHT - player_size)

rect = pygame.Rect(x, y, player_size, player_size)

return rect

player1 = get_random_spawn()

player2 = get_random_spawn()

while player1.colliderect(player2):

player2 = get_random_spawn()

portals = [(get_random_spawn(), get_random_spawn(), 0) for _ in range(num_portal_pairs)]

clock = pygame.time.Clock()

running = True

game_over = False

bullets = []

last_dir1 = (1, 0)

last_dir2 = (-1, 0)

player1_alive = True

player2_alive = True

player1_hp = 3

player2_hp = 3

def move_player(player, dx, dy):

new_rect = player.move(dx, dy)

if new_rect.left < 0 or new_rect.right > WIDTH or new_rect.top < 0 or new_rect.bottom > HEIGHT:

return player

return new_rect

while running:

screen.fill(WHITE)

now = pygame.time.get_ticks()

for event in pygame.event.get():

if event.type == pygame.QUIT:

running = False

if game_over and event.type == pygame.MOUSEBUTTONDOWN:

if button_rect.collidepoint(event.pos):

player1 = get_random_spawn()

player2 = get_random_spawn()

while player1.colliderect(player2):

player2 = get_random_spawn()

portals = [(get_random_spawn(), get_random_spawn(), 0) for _ in range(num_portal_pairs)]

bullets.clear()

game_over = False

player1_alive = True

player2_alive = True

if event.type == pygame.KEYDOWN:

if player1_alive and event.key == pygame.K_SPACE:

bullet = pygame.Rect(player1.centerx - 5, player1.centery - 5, 10, 10)

bullets.append((bullet, last_dir1, RED))

if player2_alive and event.key in (pygame.K_RETURN, pygame.K_KP_ENTER):

bullet = pygame.Rect(player2.centerx - 5, player2.centery - 5, 10, 10)

bullets.append((bullet, last_dir2, BLUE))

keys = pygame.key.get_pressed()

if not game_over:

dx1 = dy1 = 0

dx2 = dy2 = 0

if player1_alive:

if keys[pygame.K_a]: dx1 = -speed

if keys[pygame.K_d]: dx1 = speed

if keys[pygame.K_w]: dy1 = -speed

if keys[pygame.K_s]: dy1 = speed

if player2_alive:

if keys[pygame.K_LEFT]: dx2 = -speed

if keys[pygame.K_RIGHT]: dx2 = speed

if keys[pygame.K_UP]: dy2 = -speed

if keys[pygame.K_DOWN]: dy2 = speed

if dx1 or dy1: last_dir1 = (dx1 // speed, dy1 // speed)

if dx2 or dy2: last_dir2 = (dx2 // speed, dy2 // speed)

if player1_alive:

player1 = move_player(player1, dx1, dy1)

if player2_alive:

player2 = move_player(player2, dx2, dy2)

for i, (a, b, cooldown) in enumerate(portals):

if now - cooldown >= portal_cooldown:

if player1_alive and player1.colliderect(a):

player1.topleft = b.topleft

portals[i] = (a, b, now)

elif player1_alive and player1.colliderect(b):

player1.topleft = a.topleft

portals[i] = (a, b, now)

if player2_alive and player2.colliderect(a):

player2.topleft = b.topleft

portals[i] = (a, b, now)

elif player2_alive and player2.colliderect(b):

player2.topleft = a.topleft

portals[i] = (a, b, now)

if player1_alive and player2_alive and player1.colliderect(player2):

game_over = True

for a, b, cooldown in portals:

pygame.draw.rect(screen, PORTAL_COLOR, a)

pygame.draw.rect(screen, PORTAL_COLOR, b)

time_left = max(0, portal_cooldown - (now - cooldown)) // 1000

if time_left > 0:

txt = small_font.render(str(time_left), True, BLACK)

screen.blit(txt, (a.centerx - txt.get_width() // 2, a.top - 30))

screen.blit(txt, (b.centerx - txt.get_width() // 2, b.top - 30))

if player1_alive:

pygame.draw.rect(screen, RED, player1)

if player2_alive:

pygame.draw.rect(screen, BLUE, player2)

for bullet in bullets[:]:

rect, direction, color = bullet

rect.x += direction[0] * bullet_speed

rect.y += direction[1] * bullet_speed

for i, (a, b, cooldown) in enumerate(portals):

if now - cooldown >= portal_cooldown:

if rect.colliderect(a):

rect.topleft = b.topleft

portals[i] = (a, b, now)

elif rect.colliderect(b):

rect.topleft = a.topleft

portals[i] = (a, b, now)

if player1_alive and rect.colliderect(player1) and color == BLUE:

player1_hp -= 1

bullets.remove(bullet)

if player1_hp <= 0:

player1_alive = False

game_over = True

elif player2_alive and rect.colliderect(player2) and color == RED:

player2_hp -= 1

bullets.remove(bullet)

if player2_hp <= 0:

player2_alive = False

game_over = True

if not screen.get_rect().contains(rect):

bullets.remove(bullet)

continue

else:

pygame.draw.rect(screen, color, rect)

if game_over:

loser = "1" if not player1_alive else "2"

text = font.render(f"Гравець {loser} програв! HP залишилось: {player2_hp if loser == '1' else player1_hp}", True, BLACK)

screen.blit(text, ((WIDTH - text.get_width()) // 2, HEIGHT // 2 - 100))

pygame.draw.rect(screen, GRAY, button_rect)

pygame.draw.rect(screen, BLACK, button_rect, 2)

button_text = small_font.render("Грати знову", True, BLACK)

player1_hp = 3

player2_hp = 3

screen.blit(button_text, (button_rect.x + (button_width - button_text.get_width()) // 2,

button_rect.y + (button_height - button_text.get_height()) // 2))

hp_text1 = small_font.render(f"HP: {player1_hp}", True, BLACK)

hp_text2 = small_font.render(f"HP: {player2_hp}", True, BLACK)

screen.blit(hp_text1, (player1.x, player1.y - 25))

screen.blit(hp_text2, (player2.x, player2.y - 25))

pygame.display.flip()

clock.tick(60)

pygame.quit()

sys.exit()
